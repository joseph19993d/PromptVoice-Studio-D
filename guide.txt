⚙️ 5. Implementación base (React)
📄 SetupGuide.tsx
import { useState } from "react";

const steps = [
  { title: "Welcome", content: "Welcome to PromptVoice Studio" },
  { title: "Mode", content: "Choose your mode" },
  { title: "AI", content: "Connect AI" },
  { title: "TTS", content: "Connect Voice" },
  { title: "Mic", content: "Setup microphone" },
  { title: "Done", content: "You're ready!" },
];

export default function SetupGuide() {
  const [step, setStep] = useState(0);

  return (
    <div style={{ padding: 20, border: "1px solid #ccc" }}>
      <h2>{steps[step].title}</h2>
      <p>{steps[step].content}</p>

      <div style={{ marginTop: 20 }}>
        {step > 0 && (
          <button onClick={() => setStep(step - 1)}>Back</button>
        )}

        {step < steps.length - 1 ? (
          <button onClick={() => setStep(step + 1)}>
            Next
          </button>
        ) : (
          <button onClick={() => alert("Done!")}>
            Finish
          </button>
        )}
      </div>
    </div>
  );
}
